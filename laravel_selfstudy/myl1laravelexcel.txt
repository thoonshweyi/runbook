        Step 2: Create an Import Class

            Run the following command to create an import class:

            php artisan make:import PositionImport --model=Position


            =In app/Imports/PositionImport.php.

                namespace App\Imports;

                use App\Models\Position;
                use Maatwebsite\Excel\Concerns\ToModel;
                use Maatwebsite\Excel\Concerns\WithHeadingRow;

                class PositionImport implements ToModel, WithHeadingRow
                {
                    public function model(array $row)
                    {
                        return new Position([
                            'name'      => $row['name'],   // Column 'name' from Excel
                            'slug'      => $row['slug'],   // Column 'slug' from Excel
                            'status_id' => $row['status_id'], // Column 'status_id' from Excel
                            'user_id'   => $row['user_id'], // Column 'user_id' from Excel
                        ]);
                    }
                }

        Step 3: Create a Controller for Importing
            php artisan make:controller PositionsController

            =In PositionsController
                namespace App\Http\Controllers;

                use Illuminate\Http\Request;
                use Maatwebsite\Excel\Facades\Excel;
                use App\Imports\PositionImport;

                class PositionsController extends Controller
                {
    

                    public function import(Request $request)
                    {
                        $request->validate([
                            'file' => 'required|mimes:xlsx,csv,xls|max:2048'
                        ]);

                        Excel::import(new PositionImport, $request->file('file'));

                        return redirect()->back()->with('success', 'Positions Imported Successfully!');
                    }
                }
